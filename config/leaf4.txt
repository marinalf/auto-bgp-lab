feature bgp 

route-map TAG permit 10
  match tag 12345 

  router bgp auto
  router-id auto
  address-family ipv4 unicast
    redistribute direct route-map TAG
  address-family ipv6 unicast
    redistribute direct route-map TAG
  neighbor ethernet 1/1 - 2
    remote-as external
    address-family ipv4 unicast
    address-family ipv6 unicast
neighbor fe80:10:0:3::2
    remote-as internal
    address-family ipv4 unicast
      route-reflector-client
    address-family ipv6 unicast
      route-reflector-client
neighbor fe80:10:0:4::2
    remote-as internal
    address-family ipv4 unicast
      route-reflector-client
    address-family ipv6 unicast
      route-reflector-client

interface loopback0
  ip address 10.34.34.34/32 tag 12345
  ipv6 address 2001::34/128 tag 12345

  interface ethernet 1/1-2
    no switchport
    ip forward
    ipv6 link-local use-bia
    no shutdown

  interface ethernet 1/6
    description w-node3
    no switchport
    ip address 10.0.3.1/24
    ipv6 link-local fe80:10:0:3::1
    no shutdown

  interface ethernet 1/7
    description w-node4
    no switchport
    ip address 10.0.4.1/24
    ipv6 link-local fe80:10:0:4::1
    no shutdown
